# 字符串常见方法

整理了string方法，方便查阅

# 1 str.capitalize(): 首字母大写

name = "john doe"
capitalized_name = name.capitalize()
# 功能：将字符串的首字母大写
print(capitalized_name)  # 输出：'John doe'

# 2 str.casefold(): 强大的小写转换

s = "ß"
lowercased_s = s.casefold()
# 功能：将字符串转换为小写，并处理特殊字符
print(lowercased_s)  # 输出：'ss'

# 3 str.center(width[, fillchar]): 字符串居中

text = "Hello"
centered_text = text.center(10, "*")

encoded_text = text.encode(encoding='utf-8')

filename = "document.txt"
is_text_file = filename.endswith(".txt")

# 8 str.find(sub[, start[, end]]): 子字符串索引

sentence = "I have a cat."
cat_index = sentence.find("cat")
# 功能：返回子字符串在字符串中第一次出现的索引，如果不包含则返回-1

# 9 str.format(*args, **kwargs): 格式化字符串

name = "Felix"
age = 25
message = "My name is {} and I'm {} years old.".format(name, age)
# 功能：将占位符{}替换为相应的值
print(message)  # 输出：'My name is Felix and I'm 25 years old.'

# 10 str.format_map(mapping): 格式化字符串

person = {'name': 'Felix', 'age': 25}
message = "My name is {name} and I'm {age} years old.".format_map(person)
# 功能：将占位符{key}替换为映射中对应的值
print(message)  # 输出：'My name is Felix and I'm 25 years old.'
