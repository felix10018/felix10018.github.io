from selenium import webdriver
from time import sleep

from selenium.webdriver.common.by import By


class TestCase(object):
    def __init__(self):
        self.driver = webdriver.Chrome()
        self.driver.get('http://www.baidu.com')
        self.driver.maximize_window()
        sleep(1)
    def test_id(self):
        # id是唯一的；id是唯一的
        element = self.driver.find_element_by_id('kw')
        element.send_keys('selenium')
        print(type(element))
        # <class 'selenium.webdriver.remote.webelement.WebElement'> 网页元素
        self.driver.find_element_by_id('su').click()
        sleep(3)
        # self.driver.quit()

    def test_name(self):
        # find_element_by_name 方法可能返回多个元素，返回第一个
        # self.driver.find_elements_by_name() 返回一个集合
        self.driver.find_element_by_name('wd').send_keys('selenium')
        self.driver.find_element_by_id('su').click()
        sleep(3)
        #self.driver.quit()
    def test_linktext(self):
        self.test_id()
        self.driver.find_element_by_link_text('百度首页').click()
        sleep(3)

    def test_partial_link_text(self):

        self.test_id()
        self.driver.find_element_by_partial_link_text('首页').click()
        sleep(3)

    def test_xpath(self):
        self.driver.find_element_by_xpath('//*[@id="kw"]').send_keys('web自动化测试')
        self.driver.find_element_by_id('su').click()
        sleep(3)

    def test_tag(self):
        input = self.driver.find_element_by_tag_name('input')[0]
        print(input)

    def test_css_selector(self):
        self.driver.find_element_by_css_selector('#kw').send_keys('web自动化测试')
        self.driver.find_element_by_id('su').click()
        sleep(3)

    def test_class_name(self):
        self.driver.find_element_by_class_name('s_ipt').send_keys('web自动化测试')
        self.driver.find_element_by_id('su').click()
        sleep(3)
        self.driver.quit()

    def test_all(self):
        self.driver.find_element(By.ID,value='kw').send_keys('selenium')
        self.driver.find_element_by_id('su').click()
        sleep(3)
        self.driver.quit()

from selenium import webdriver
from time import sleep


class TestCase(object):
    def __init__(self):
        self.driver = webdriver.Chrome()
        self.driver.get('http://www.baidu.com')
        self.driver.maximize_window()
    def test_prop(self):
        print(self.driver.name)# 浏览器名称
        print(self.driver.current_url) # 访问的当前网页url
        print(self.driver.title) #获取当前网页标题
        print(self.driver.window_handles) #当前网页源码
        print(self.driver.page_source)  #窗口句柄
        #self.driver.quit()

    def test_method(self):
        self.driver.find_element_by_id('kw').send_keys('selenium')
        self.driver.find_element_by_id('su').click()
        sleep(2)
        self.driver.back()  #网页后退
        sleep(2)
        self.driver.refresh() #网页刷新
        sleep(2)
        self.driver.forward() #网页前进

        self.driver.close() # 只关闭当前tab
        sleep(3)
        self.driver.quit() # 关闭浏览器

    def test_windows(self):
        title = '百度新闻——海量中文资讯平台'
        self.driver.find_element_by_link_text('新闻').click()
        windows = self.driver.window_handles
        print(windows)

        ##如何切换窗口
        while 1:
            for w in windows:
                self.driver.switch_to.window(w)
                sleep(2)


if __name__ == '__main__':
    case = TestCase()

九、3种等待方法
from selenium import webdriver 
from time import sleep
from selenium.webdriver.support import expected_conditions as EC
from selenium.webdriver.support.wait import WebDriverWait


class TestCase(object):
    def __init__(self):
        self.driver = webdriver.Chrome()
        self.driver.get('http://www.baidu.com')
        # sleep(2)

    def test_sleep(self): #固定等待
        self.driver.find_element_by_id('kw').send_keys('selenium')
        # sleep(2) # 线程阻塞 blocking wait
        self.driver.find_element_by_id('su').click()
        # sleep(3)
        self.driver.quit()

    def test_implicitly(self): #隐式等待
        self.driver.implicitly_wait(10)  ##隐式等待10s,如果同时设置隐式等待和固定等待，那哪个时间长以哪个为准
        self.driver.find_element_by_id('kw').send_keys('selenium')
        # sleep(2) # 线程阻塞 blocking wait
        self.driver.find_element_by_id('su').click()
        # sleep(3)
        self.driver.quit()

    def test_wait(self):
        wait = WebDriverWait(self.driver,2)  ##显式等待，项目常用
        wait.until(EC.title_is('百度一下，你就知道'))
        self.driver.find_element_by_id('kw').send_keys('selenium')
        # sleep(2) # 线程阻塞 blocking wait
        self.driver.find_element_by_id('su').click()
        # sleep(3)
        self.driver.quit()

if __name__ == '__main__':
    case = TestCase()
    # case.test_sleep()
    # case.test_implicitly()
    case.test_wait()

#元素等待条件
from selenium import webdriver 
from time import sleep
import os

from selenium.webdriver.common.by import By
from selenium.webdriver.support import expected_conditions as EC

from selenium.webdriver.support.wait import WebDriverWait


class TestCase(object):
    def __init__(self):
        self.driver = webdriver.Chrome()
        path = 'file:///'+os.path.abspath('等待条件.html')
        self.driver.get(path)

    def test(self):
        self.driver.find_element_by_id('btn').click()
        # 显示等待
        wait = WebDriverWait(self.driver, 4)
        wait.until(EC.text_to_be_present_in_element((By.ID,'id2'),'id 2'))
        print(self.driver.find_element_by_id('id2').text)
        print('ok')

if __name__ == '__main__':
    case = TestCase()
    case.test()
